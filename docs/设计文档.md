## 项目结构

```
ikun-cloud/
├── server/                  # Bun.js 后端
│   └── src/
│       ├── index.ts         # 入口
│       ├── db/              # 数据库 (SQLite + Drizzle)
│       ├── routes/          # API 路由
│       ├── services/        # 业务逻辑
│       ├── middleware/       # JWT 认证、错误处理
│       └── utils/           # 工具函数
├── web/                     # Vue 3 前端
│   └── src/
│       ├── views/           # 页面 (登录、仪表盘、VM 列表、VM 详情)
│       ├── api/             # 请求封装
│       ├── stores/          # Pinia 状态管理
│       ├── router/          # 路由配置
│       └── layouts/         # 布局组件
├── ikun-ctl/                # CLI 工具 (Python)
│   └── ikun-ctl.py
├── build-rootfs-v2.sh       # 基础镜像构建脚本
└── README.md
```

## 后端 API

```
# 公开接口（无需 token）
POST   /api/public/login         # 登录 (返回 JWT)
POST   /api/public/logout        # 登出
POST   /api/public/register      # 注册
GET    /api/public/site-info     # 站点信息
POST   /api/public/cap/challenge # Cap 人机验证挑战
POST   /api/public/cap/redeem    # Cap 验证兑换

# 租户接口（root + user）
GET    /api/user/me               # 当前用户信息
PUT    /api/user/password         # 修改密码
GET    /api/user/vms              # 我的 VM 列表
GET    /api/user/vms/:id          # VM 详情
POST   /api/user/vms/:id/start    # 启动
POST   /api/user/vms/:id/stop     # 停止
POST   /api/user/vms/:id/restart  # 重启
POST   /api/user/vms/:id/reset-password  # 重置密码
POST   /api/user/vms/:id/reinstall      # 重装系统
GET    /api/user/network/vms/:id/ports       # 端口映射列表
POST   /api/user/network/vms/:id/ports       # 添加端口映射
DELETE /api/user/network/vms/:id/ports/:pid  # 删除端口映射

# 管理员接口（仅 root）
GET    /api/admin/dashboard       # 仪表盘
GET    /api/admin/users           # 用户列表
POST   /api/admin/users           # 创建用户
PUT    /api/admin/users/:id       # 修改用户
DELETE /api/admin/users/:id       # 删除用户
GET    /api/admin/vms             # 所有 VM 列表
POST   /api/admin/vms             # 创建 VM
GET    /api/admin/vms/:id         # VM 详情
POST   /api/admin/vms/:id/start   # 启动
POST   /api/admin/vms/:id/stop    # 停止
POST   /api/admin/vms/:id/restart # 重启
DELETE /api/admin/vms/:id         # 删除
POST   /api/admin/vms/:id/assign  # 分配给用户
POST   /api/admin/vms/:id/unassign # 取消分配
GET    /api/admin/images          # 镜像列表
GET    /api/admin/settings        # 系统设置
PUT    /api/admin/settings        # 更新设置
```

> 除 `/api/public/*` 外，所有接口需携带 `Authorization: Bearer <token>`

## 数据库设计

```sql
-- 管理员用户
CREATE TABLE users (
  id            INTEGER PRIMARY KEY AUTOINCREMENT,
  username      TEXT NOT NULL UNIQUE,
  password_hash TEXT NOT NULL,
  role          TEXT NOT NULL DEFAULT 'user',  -- 'root' | 'user'
  invite_code   TEXT,                          -- 使用的邀请码
  created_at    TEXT NOT NULL DEFAULT (datetime('now'))
);

-- VM 实例
CREATE TABLE vms (
  id            TEXT PRIMARY KEY,        -- xd8j4
  name          TEXT NOT NULL,           -- 小鸡-张三
  status        TEXT NOT NULL DEFAULT 'stopped',
  cpus          INTEGER NOT NULL DEFAULT 1,
  memory_mb     INTEGER NOT NULL DEFAULT 512,
  disk_gb       INTEGER NOT NULL DEFAULT 20,
  base_image    TEXT NOT NULL,           -- 基础镜像名 (debian12-custom)
  ip            TEXT NOT NULL,           -- 内网 IP: 10.100.0.2
  mac           TEXT NOT NULL,
  tap           TEXT NOT NULL,
  ssh_port      INTEGER NOT NULL,
  password      TEXT,
  api_socket    TEXT,
  owner_id      INTEGER REFERENCES users(id) ON DELETE SET NULL,  -- 租户
  created_at    TEXT NOT NULL DEFAULT (datetime('now')),
  updated_at    TEXT NOT NULL DEFAULT (datetime('now'))
);

-- 端口映射
CREATE TABLE port_forwards (
  id            INTEGER PRIMARY KEY AUTOINCREMENT,
  vm_id         TEXT NOT NULL REFERENCES vms(id) ON DELETE CASCADE,
  host_port     INTEGER NOT NULL,
  guest_port    INTEGER NOT NULL,
  protocol      TEXT NOT NULL DEFAULT 'tcp',
  created_at    TEXT NOT NULL DEFAULT (datetime('now'))
);

-- 邀请码
CREATE TABLE invite_codes (
  id            INTEGER PRIMARY KEY AUTOINCREMENT,
  code          TEXT NOT NULL UNIQUE,
  remark        TEXT NOT NULL,
  used_by       INTEGER REFERENCES users(id) ON DELETE SET NULL,
  created_at    TEXT NOT NULL DEFAULT (datetime('now')),
  used_at       TEXT
);

-- 网站配置（key-value）
CREATE TABLE site_settings (
  key           TEXT PRIMARY KEY,
  value         TEXT NOT NULL
);

-- 公告
CREATE TABLE announcements (
  id            INTEGER PRIMARY KEY AUTOINCREMENT,
  title         TEXT NOT NULL,
  content       TEXT NOT NULL,
  is_active     INTEGER NOT NULL DEFAULT 1,
  created_at    TEXT NOT NULL DEFAULT (datetime('now')),
  updated_at    TEXT NOT NULL DEFAULT (datetime('now'))
);

-- 监控采样（保留 24 小时）
CREATE TABLE traffic_samples (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  vm_id TEXT NOT NULL REFERENCES vms(id) ON DELETE CASCADE,
  rx_bytes INTEGER NOT NULL, tx_bytes INTEGER NOT NULL,
  sampled_at TEXT NOT NULL DEFAULT (datetime('now'))
);
CREATE TABLE cpu_samples (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  vm_id TEXT NOT NULL REFERENCES vms(id) ON DELETE CASCADE,
  usage REAL NOT NULL,
  sampled_at TEXT NOT NULL DEFAULT (datetime('now'))
);
CREATE TABLE mem_samples (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  vm_id TEXT NOT NULL REFERENCES vms(id) ON DELETE CASCADE,
  used INTEGER NOT NULL, total INTEGER NOT NULL,
  sampled_at TEXT NOT NULL DEFAULT (datetime('now'))
);
CREATE TABLE disk_samples (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  vm_id TEXT NOT NULL REFERENCES vms(id) ON DELETE CASCADE,
  read_bytes INTEGER NOT NULL, write_bytes INTEGER NOT NULL,
  sampled_at TEXT NOT NULL DEFAULT (datetime('now'))
);

-- 小时聚合（永久保留）
CREATE TABLE vm_metrics_hourly (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  vm_id TEXT NOT NULL REFERENCES vms(id) ON DELETE CASCADE,
  hour_start TEXT NOT NULL,
  rx_bytes INTEGER NOT NULL DEFAULT 0, tx_bytes INTEGER NOT NULL DEFAULT 0,
  cpu_avg REAL NOT NULL DEFAULT 0, cpu_max REAL NOT NULL DEFAULT 0,
  mem_avg INTEGER NOT NULL DEFAULT 0, mem_max INTEGER NOT NULL DEFAULT 0, mem_total INTEGER NOT NULL DEFAULT 0,
  disk_read INTEGER NOT NULL DEFAULT 0, disk_write INTEGER NOT NULL DEFAULT 0
);
```

## VM 创建流程

```
1. 选择基础镜像 (debian12-custom.raw)
2. qemu-img 创建 qcow2 磁盘 (基于镜像 COW)
3. 挂载磁盘，写入网络配置和密码
4. 创建 TAP 网络设备
5. 添加 iptables 端口映射
6. 启动 Cloud Hypervisor
7. 保存配置到数据库
```
